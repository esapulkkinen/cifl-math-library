 instance (Num a) => Num (Dual m a) where
   (Dual f) + (Dual g) = Dual $ \x -> f x + g x
   (Dual f) - (Dual g) = Dual $ \x -> f x - g x
   (Dual f) * (Dual g) = Dual $ \x -> f x * g x
   negate (Dual f) = Dual $ \x -> negate (f x)
   abs (Dual f) = Dual $ \x -> abs (f x)
   signum (Dual f) = Dual $ \x -> signum (f x)
   fromInteger i = Dual $ const (fromInteger i)

>(*!) :: (Scalar (U d) ~ Double, (Scalar (U d') ~ Double),
>         Unit (U d), Unit (U d'), Unit (U (DPlus d d')),
>         Scalar (U (DPlus d d')) ~ Double)
>        => U d -> U d' -> U (DPlus d d')
>(*!) x y = fromAmount (amount x * amount y)

>(/!) :: (Scalar (U d) ~ Double, Scalar (U d') ~ Double

>         (Double ~ Scalar (U (DMinus d d'))))
>        => U d -> U d' -> U (DMinus d d')
>(/!) x y = fromAmount (amount x / amount y)

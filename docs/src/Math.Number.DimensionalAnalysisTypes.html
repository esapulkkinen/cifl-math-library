>import GHC.Types (Type)

>import Data.Ratio ((%))
>import Data.Proxy
>import Math.Matrix.Interface
>import qualified Math.Number.DimensionalAnalysis as DA

>data Quantity :: Dimension -> Type where
>   Value :: Double -> Quantity dim

>instance VectorSpace (Quantity dim) where
>   type Scalar (Quantity dim) = Double
>   vzero = Value vzero
>   vnegate (Value x) = Value (vnegate x)
>   (Value x) %+ (Value y) = Value (x %+ y)
>   x %* (Value y) = Value (x %* y)

>mapQuantity :: (Double -> Double) -> Quantity dim -> Quantity dim
>mapQuantity f (Value x) = Value (f x)

>type DimensionNat l w t c te lu su = (KnownNat l, KnownNat w, KnownNat t,
>                                       KnownNat c, KnownNat te, KnownNat lu,
>                                       KnownNat su)

>toRuntime :: (DimensionNat l w t c te lu su)
>   => Quantity ('Dimension l w t c te lu su) -> DA.Dimension
>toRuntime (Value x :: Quantity ('Dimension l w t c te lu su)) = DA.Dimension
> (fromIntegral $ natVal (Proxy :: Proxy l))
> (fromIntegral $ natVal (Proxy :: Proxy w))
> (fromIntegral $ natVal (Proxy :: Proxy t))
> (fromIntegral $ natVal (Proxy :: Proxy c))
> (fromIntegral $ natVal (Proxy :: Proxy te))
> (fromIntegral $ natVal (Proxy :: Proxy lu))
> (fromIntegral $ natVal (Proxy :: Proxy su))

>data Dimension = Dimension {
>   length_power :: Nat,
>   weight_power :: Nat,
>   time_power :: Nat,
>   current_power :: Nat,
>   temperature_power :: Nat,
>   luminosity_power :: Nat,
>   substance_power :: Nat }

>type Dimensionless = 'Dimension 0 0 0 0 0 0 0
>type Meter =     'Dimension 1 0 0 0 0 0 0
>type Kilogram =  'Dimension 0 1 0 0 0 0 0
>type Second =    'Dimension 0 0 1 0 0 0 0
>type Ampere =   'Dimension 0 0 0 1 0 0 0
>type Kelvin =   'Dimension 0 0 0 0 1 0 0
>type Candela =  'Dimension 0 0 0 0 0 1 0
>type Mole   =     'Dimension 0 0 0 0 0 0 1

>instance (DimensionNat l w t c te lu su)
> => Show (Quantity ('Dimension l w t c te lu su)) where
>   show z@(Value x) = show x ++ " " ++ show (toRuntime z)

>meter :: Quantity Meter
>meter = Value 1.0
>
>second :: Quantity Second
>second = Value 1.0

>kilogram :: Quantity Kilogram
>kilogram = Value 1.0

>ampere :: Quantity Ampere
>ampere = Value 1.0

>kelvin :: Quantity Kelvin
>kelvin = Value 1.0

>candela :: Quantity Candela
>candela = Value 1.0

>mole :: Quantity Mole
>mole = Value 1.0

>type family Plus (a :: Dimension) (b :: Dimension) :: Dimension
>type instance Plus ('Dimension l w t c te lu su) ('Dimension l2 w2 t2 c2 te2 lu2 su2) = 'Dimension (l + l2) (w + w2) (t + t2) (c + c2) (te + te2) (lu + lu2) (su + su2)

>type family Minus (a :: Dimension) (b :: Dimension) :: Dimension
>type instance Minus ('Dimension l w t c te lu su) ('Dimension l2 w2 t2 c2 te2 lu2 su2) = 'Dimension (l - l2) (w - w2) (t - t2) (c - c2) (te - te2) (lu - lu2) (su - su2)

>type family DMul (a :: Dimension) (b :: Nat) :: Dimension
>type instance DMul ('Dimension l w t c te lu su) (b :: Nat) = 'Dimension (b * l) (b * w) (b * t) (b * c) (b * te) (b * lu) (b * su)

>type family DDiv (a :: Dimension) (b :: Nat) :: Dimension
>type instance DDiv ('Dimension l w t c te lu su) (b :: Nat) = 'Dimension (Div l b) (Div w b) (Div t b) (Div c b) (Div te b) (Div lu b) (Div su b)

>class ProductDimension (a :: Dimension) (b :: Dimension) where
>   (*%) :: Quantity a -> Quantity b -> Quantity (Plus a b)
>   (/%) :: Quantity a -> Quantity b -> Quantity (Minus a b)

>dimensionless :: Double -> Quantity Dimensionless
>dimensionless = Value

>class FloatingDimension (a :: Dimension) where
>   square :: Quantity a -> Quantity (DMul a 2)
>   sqroot :: Quantity a -> Quantity (DDiv a 2)

>instance ProductDimension a b where
>   (Value x) *% (Value y) = Value (x*y)
>   (Value x) /% (Value y) = Value (x/y)

>instance FloatingDimension a where
>   square (Value x) = Value (x*x)
>   sqroot (Value x) = Value (sqrt x)

>instance Num (Quantity dim) where
>   (Value x) + (Value y) = Value (x + y)
>   (Value x) - (Value y) = Value (x - y)
>   (Value x) * (Value y) = error "Invalid dimensions, please use *% for multiplication"
>   negate (Value x) = Value (negate x)
>   abs (Value x) = Value (abs x)
>   signum (Value x) = Value (signum x)
>   fromInteger x = Value (fromInteger x)
